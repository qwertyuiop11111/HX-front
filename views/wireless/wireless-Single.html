	{
        chart:{
            backgroundColor:'#fff',
            margin:[-3,0,25,51],
            animation:false
        },
        credits:{
            enabled:false
        },
        title: {
            text: null
        },
        xAxis: {
            title: {
                text: '时间',
                align:'high',
                style:{"fontSize": "15px","color": "#7a8282"}
            },
            lineWidth: 1,
            lineColor:'#8a8a8a',
            labels:{
                enabled:false
            },
            tickLength:0,
            crosshair: {
                width: 2,
                color: '#ff9666',
                dashStyle: 'Solid'
            }
        },
        yAxis: {
            title: {
                text: '电平(dBμV)',
                align:'high',
                style:{"fontSize": "15px","color": "#7a8282"},
                x:10,
                y:10
            },
            gridLineColor:'#8a8a8a',
            gridLineDashStyle:'ShortDash',
            labels:{
                style:{"fontSize": "15px","color": "#7a8282"},
                x:-5
            },
            tickPositions:[-20,0,20,40,60,80,100],
            lineWidth: 1,
            lineColor:'#8a8a8a'
        },
        legend: {
            enabled: false
        },
        plotOptions: {
            line: {
                color:'#15ba57',
                marker: {
                    radius: 2
                },
                lineWidth: 2,
                states: {
                    hover: {
                        lineWidth: 2
                    }
                },
                threshold: null
            }
        },
        tooltip: {
            formatter: function () {
                // var s = '<b>时间: ' + this.x + 's</b><br/>';
                var s='';
                $.each(this.points, function () {
                    s += '' + "电平" + ': ' +
                        this.y + 'dBμV';
                });
                return s;
            },
            shared: true
        },
        series: [{
            type: 'line',
            name: '电平',
